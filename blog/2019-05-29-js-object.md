- js 的对象

#### 对象创建

```javascript
//定义对象 json对象
var obj = {
  name: "dfx",
  age: 18,
  pay: function () {
    console.log("挣钱");
  },
};
console.log(obj);

// 定义2
var obj2 = new Object();
obj2.name = "dmy";
obj2.age = "16";
obj2.pay = function () {
  console.log("花钱");
};
console.log(obj2);

// 省略key
var username = "dd";
var age = 20;

var obj3 = {
  username,
  age,
  pay() {
    console.log("加班");
  },
};

console.log(obj3.age);

// 构造函数创建对象
function Person(name, age) {
  this.name = name;
  this.age = age;
}

var p1 = new Person("dfx", 29);
var p2 = new Person("dmy", 20);
console.log(p1, p2);
```

##### 遍历对象

```javascript
var obj = {
  name: "dfx",
  age: 18,
  pay: function () {
    console.log("挣钱");
  },
};

for (key in obj) {
  console.log(key, "--", obj[key]);
}
```

##### 删除对象属性

```javascript
delete obj3.pay;
//无法访问到对象属性  TypeError: obj3.pay is not a function
console.log(obj3.pay());
}
```
