#### 选择排序

- 选择排序：双层循环寻找最小值
```javascript
let arr = [4,2,3,6,1,5];

for(let i = 0; i<arr.length; i++){
  // 标记最小值的下标 最开始为0
  let mainPoint = i;
  //获取最小值的下标
  for(let j=i+1; j<arr.length;j++){
      if(arr[j] < arr[mainPoint]){
          mainPoint = j;
      }
  }
  console.log(mainPoint)
  //交换值 tmp:临时中间变量用于交换
  let tmp = arr[mainPoint];
  arr[mainPoint] = arr[i];
  arr[i] = tmp;
}

console.log(arr);
```
